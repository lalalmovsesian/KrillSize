Analyses

Testing interannual variability using Linear Mixed Models because:

  • nested data (no need to collapse data to means)

  • not independent data (avoiding pseudoreplication and inflated significance)

  • considering uneven sampling effort

  • more suitable for different months and different numbers of cruises in year

In this model, year is used as fixed effect and cruise/jar as random factors, therefore testing if variability among years is larger than within-jar/cruise.

downsides: Some years have more cruises/samples than others, which can influence the precision Years are not considered directional, and this model will not show overall trends. Considering that the main goal is to identify changes in response to climate anomalies, this is not a problem.

Size Calculation

Adding columns for width and length midpoint for each size range in each type:

df <- df %>%
  mutate(length_mid = (length_l + length_h) / 2) %>%
  mutate(width_mid = (width_l + width_h) / 2) %>%
relocate(length_mid, .after = length_h) %>%
  relocate(width_mid, .after = width_h)

Calculating volume per each row by considering krill cone-shaped, using width_mid as diameter and length_mid as height:

df <- df %>%
  mutate(volume = (1/3) * pi * (width_mid / 2)^2 * length_mid) %>%
  relocate(volume, .after = width_mid) %>%
  filter(!is.na(volume))

Calculating mean volume per jar: (weighted mean of volume instead of means of l and w first then calculating mean volume with mean l and w)

df_vol <- df %>%
  group_by(species_code, jar, type, base_date, moci) %>%
  filter(!is.na(tot_num)) %>%
  summarise(volume_weighted = sum(volume * tot_num) / sum(tot_num),
    .groups = "drop")

plot_adult <- function(data, species_name = "Adult") {

  data <- data %>%
    mutate(year = year(base_date))
  
  summary_vol <- data %>%
    group_by(year, type) %>%
    summarise(volume_weighted = mean(volume_weighted, na.rm = TRUE),
              .groups = "drop")

  summary_moci <- data %>%
    group_by(year) %>%
    summarise(moci = mean(moci, na.rm = TRUE),
              .groups = "drop")

  vol_min <- min(summary_vol$volume_weighted, na.rm = TRUE)
  vol_max <- max(summary_vol$volume_weighted, na.rm = TRUE)
  vol_mid <- mean(c(vol_min, vol_max))

  moci_max <- max(abs(summary_moci$moci), na.rm = TRUE)
  scale_factor <- (vol_max - vol_min) / (2 * moci_max)

  ggplot() +

    # Volume (viridis, consistent)
    geom_line(data = summary_vol,
              aes(x = factor(year), y = volume_weighted,
                  color = type, group = type),
              linewidth = 1) +

    geom_point(data = summary_vol,
               aes(x = factor(year), y = volume_weighted,
                   color = type),
               size = 3) +

    # MOCI (neutral + dashed ALWAYS)
    geom_line(data = summary_moci,
              aes(x = factor(year),
                  y = (moci * scale_factor) + vol_mid,
                  group = 1),
              color = "grey30",
              linewidth = 1.2,
              linetype = "dashed") +

    scale_y_continuous(
      name = "Weighted mean body volume (mm3)",
      sec.axis = sec_axis(~ (. - vol_mid) / scale_factor,
                          name = "MOCI")) +

    scale_color_viridis_d(option = "D", end = 0.8) +

labs(
  x = "Year",
  color = "Type",
  title = bquote(italic(.(species_name)))
) +

    theme_minimal(base_size = 13) +
    theme(
      panel.grid.minor = element_blank(),
      axis.text.x = element_text(angle = 45, hjust = 1)
    )
}

plot_adultm_species_compare <- function(data, sex_filter = "m", title_label = "Adults") {

  data <- data %>%
    mutate(year = year(base_date)) %>%
    filter(type == sex_filter)

  # Volume summary (per species per year)
  summary_vol <- data %>%
    group_by(year, species_code) %>%
    summarise(volume_weighted = mean(volume_weighted, na.rm = TRUE),
              .groups = "drop")

  # MOCI summary (shared across species)
  summary_moci <- data %>%
    group_by(year) %>%
    summarise(moci = mean(moci, na.rm = TRUE),
              .groups = "drop")

  # Scaling (same axis)
  vol_min <- min(summary_vol$volume_weighted, na.rm = TRUE)
  vol_max <- max(summary_vol$volume_weighted, na.rm = TRUE)
  vol_mid <- mean(c(vol_min, vol_max))

  moci_max <- max(abs(summary_moci$moci), na.rm = TRUE)
  scale_factor <- (vol_max - vol_min) / (2 * moci_max)

  ggplot() +

    # Species lines (viridis)
    geom_line(data = summary_vol,
              aes(x = factor(year),
                  y = volume_weighted,
                  color = species_code,
                  group = species_code),
              linewidth = 1) +

    geom_point(data = summary_vol,
               aes(x = factor(year),
                   y = volume_weighted,
                   color = species_code),
               size = 3) +

    # MOCI (shared, dashed)
    geom_line(data = summary_moci,
              aes(x = factor(year),
                  y = (moci * scale_factor) + vol_mid,
                  group = 1),
              color = "grey30",
              linewidth = 1.2,
              linetype = "dashed") +

    scale_y_continuous(
      name = "Weighted mean body volume (mm3)",
      sec.axis = sec_axis(~ (. - vol_mid) / scale_factor,
                          name = "MOCI")) +

    scale_color_viridis_d(option = "D", end = 0.85) +

    labs(
      x = "Year",
      color = "Species",
      title = paste(title_label, "-", sex_filter)) +

    theme_minimal(base_size = 13) +
    theme(
      panel.grid.minor = element_blank(),
      axis.text.x = element_text(angle = 45, hjust = 1))
}
plot_adultm_species_compare(df_vol)
