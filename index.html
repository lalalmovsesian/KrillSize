Filtering data

Extracting year and month information from the cruise column and converting it to a convenient format. Creating base_date in the yyyy-mm-01 format to make sure it is treated as date in R. Used “cruise” column as reference rather than “date”, since some sampling is done early in the next month (e.g. sampling date is 7/1 but it’s from june cruise)

Size Calculation

Adding columns for width and length midpoint for each size range in each type:

df <- data_day%>%
  mutate(length_mid = (length_l + length_h) / 2) %>%
  mutate(width_mid = (width_l + width_h) / 2) %>%
relocate(length_mid, .after = length_h) %>%
  relocate(width_mid, .after = width_h)

Calculating volume per each row by considering krill cone-shaped, using width_mid as diameter and length_mid as height:

df <- df %>%
  mutate(volume = (1/3) * pi * (width_mid / 2)^2 * length_mid) %>%
  relocate(volume, .after = width_mid) %>%
  filter(!is.na(volume))

Calculating mean volume per jar: (weighted mean of volume instead of means of l and w first then calculating mean volume with mean l and w)

df_vol <- df %>%
  group_by( jar, type, line, cruise, variable, value, index, lag, depth_mean) %>%
  filter(!is.na(tot_num)) %>%
  summarise(n = sum(tot_num), volume_weighted = sum(volume * tot_num) / n,
    .groups = "drop")

#Visuals

df_vol <- df_vol %>%
  mutate(cruise = factor(cruise, levels = sort(unique(cruise))))
df_volume <- df_vol %>%
  group_by(cruise, variable, value) %>%
  summarise(volume_weighted = mean(volume_weighted, na.rm = TRUE), .groups = "drop")

vol_min <- min(df_volume$volume_weighted, na.rm = TRUE)
vol_max <- max(df_volume$volume_weighted, na.rm = TRUE)

df_plot_scaled <- df_volume %>%
  group_by(variable) %>%
  mutate(var_min = min(value, na.rm = TRUE),
    var_max = max(value, na.rm = TRUE),
    value_scaled = ifelse(var_max == var_min,
      (vol_min + vol_max) / 2,  
      vol_min + (value - var_min) * (vol_max - vol_min) / (var_max - var_min))) %>%
  ungroup() %>%
  select(-var_min, -var_max)


p <- plot_ly()

Spearman rank correlaion

(data not normal)

df_plot <- df_vol %>%
  select(cruise, variable, value, volume_weighted)

results <- df_plot %>%
  
  group_by(variable) %>%
  summarise(cor = cor(value, volume_weighted, method = "spearman", use = "complete.obs"),
    p_value = cor.test(
      value, volume_weighted,
      method = "spearman",
      exact = FALSE
    )$p.value,
    n = sum(complete.cases(value, volume_weighted)),
    .groups = "drop") %>%
  
  mutate(p_adj = p.adjust(p_value, method = "BH")) %>%
  arrange(p_adj)
results %>%
  mutate(sig = p_adj < 0.05,
    variable = reorder(variable, cor)) %>%
  
  ggplot(aes(x = cor, y = variable, color = sig)) +
  geom_point(size = 3) +
  geom_vline(xintercept = 0, linetype = "dashed") +
  scale_color_manual(values = c("TRUE" = "#1b9e77", "FALSE" = "#d95f02"),
    labels = c("FALSE" = "Not significant", "TRUE" = "Significant")) +
  
  annotate("text", x = -0.25, y = Inf, label = "Negative", vjust = 1.5) +
  annotate("text", x = 0.25, y = Inf, label = "Positive", vjust = 1.5) +
  
  labs(x = "Spearman correlation with volume_weighted") +
  theme_minimal()
